git init --  começa tudo novo
git add. -- prepara os arquivs pra comit
git commit -m "versão 1.0" --- pra comentar e salavar
git status --- mostra se ta tudo ok ou faltou entrar uma nova atualização
git add nomedoarquivo.txt --- commita somente o arquvio selecionado
git log --- mostra o historico dos commit e diz as informações
git diff --- mostra no console as difereças das alterações recem feitas
git restore nomedoarquivo.txt --- desfaz as alterações feitas no documento
git branch --- me diz qual raificação eu estou
git checkout -b ramificaçao-dois --- cria uma nova ramificação e me move pra ela
git checkout master --- eu volto pra ramificação principal ou para outra depedendo do nome
git merge ramificaçao-dois --- estando numa ramificaçao 1 ou 2 ele mescla com as alterações de outra ramificação depedendo do nome
